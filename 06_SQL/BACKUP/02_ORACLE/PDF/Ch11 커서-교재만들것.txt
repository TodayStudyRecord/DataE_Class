select * from usertbl;

declare 
    userid usertbl.userid%TYPE;
    birthyear usertbl.birthyear%TYPE;
    CURSOR CUR_01
    IS
    select 
    userid,birthyear 
    from 
    usertbl;
begin
 open CUR_01;   -- 커서열기
 Loop
    Fetch CUR_01 into userid,birthyear;   -- 커서에서 데이터 가져오기
    exit when CUR_01 %NOTFOUND; --커서가 없을 경우 종료
    DBMS_OUTPUT.PUT_LINE(userid||' '||birthyear);
 end Loop;
 CLOSE CUR_01;  --커서닫기
end;

declare 
    userid usertbl.userid%TYPE;
    birthyear usertbl.birthyear%TYPE;
    CURSOR CUR_01
    IS
    select 
    userid,birthyear 
    from 
    usertbl;
begin
 open CUR_01;   -- 커서열기
 Loop
    Fetch CUR_01 into userid,birthyear;   -- 커서에서 데이터 가져오기
    exit when CUR_01 %NOTFOUND; --커서가 없을 경우 종료
    if birthyear>=1970 then
        DBMS_OUTPUT.PUT_LINE(userid||' '||birthyear);
    end if;
 end Loop;
 CLOSE CUR_01;  --커서닫기
end;
